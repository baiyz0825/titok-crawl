## 三、可靠的点击方法

### 方法 1: 使用 querySelector (推荐)
```javascript
// 输入验证码
const input = document.querySelector('#button-input');
input.value = '123456';
input.dispatchEvent(new Event('input', { bubbles: true }));

// 点击验证按钮
const verifyBtn = document.querySelector('.uc_verification_component_btn-EQNDAT.primary');
verifyBtn.click();
```

### 方法 2: 使用文本查找
```javascript
// 查找包含"验证"文本的元素
const elements = document.querySelectorAll('*');
const verifyBtn = Array.from(elements).find(el =>
  el.textContent?.trim() === '验证'
);
verifyBtn?.click();
```

### 方法 3: 模拟点击事件
```javascript
const verifyBtn = document.querySelector('.uc_verification_component_btn-EQNDAT.primary');
verifyBtn.dispatchEvent(new MouseEvent('click', {
  bubbles: true,
  cancelable: true,
  view: window
}));
```

## 四、Playwright 实现

```python
async def input_verification_code(page, code: str):
    """输入验证码"""
    # 等待输入框聚焦
    await page.wait_for_selector('#button-input')

    # 输入验证码
    await page.fill('#button-input', code)

    # 等待按钮可点击（移除 disabled 属性）
    await page.wait_for_selector(
        'div.uc_verification_component_btn-EQNDAT.primary:not(.disabled-AsTepp)',
        timeout=5000
    )

    # 点击验证按钮
    await page.click('div.uc_verification_component_btn-EQNDAT.primary')

    # 等待验证结果
    await page.wait_for_selector(
        'text="验证成功"',
        timeout=10000
    )
```
